    <script>

        const { createApp, ref, onMounted } = Vue;

        const SUPABASE_URL = 'https://zapiekidlfpyjvlvnrhc.supabase.co';

        const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InphcGlla2lkbGZweWp2bHZucmhjIiwicm9sZSI6ImFub24iLCJpYXQiOjE3ODA0OTE3NTMsImV4cCI6MjA5NjA2Nzc1M30.1QGzAJQMj45KSyiFfuP_EPNqcKgoPMQO3aplqvn1iOc';

        // ------------------------------------------


        const supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);


        createApp({

            setup() {

                const currentTab = ref('workflow');

                const loading = ref(false);

                const config = ref({ workflow_text: '', workflow_images: [], schedules: [] });

                const logs = ref([]);

                const editLog = ref(null);

                const tabs = [

                    { id: 'workflow', name: '工作流程' },

                    { id: 'schedule', name: '排班计划' },

                    { id: 'logs', name: '工作日志' },

                    { id: 'issues', name: '问题汇总' }

                ];


                const fetchData = async () => {

                    loading.value = true;

                    try {

                        // 1. 获取配置

                        const { data: cfg } = await supabase.from('project_config').select('*').eq('id', 1).single();

                        if (cfg) config.value = cfg;


                        // 2. 获取日志 (按日期倒序)

                        const { data: lg } = await supabase.from('daily_logs').select('*').order('date', { ascending: false });

                        logs.value = lg || [];

                    } catch (err) {

                        console.error('Fetch error:', err);

                    } finally {

                        loading.value = false;

                    }

                };


                const saveConfig = async () => {

                    loading.value = true;

                    const { error } = await supabase.from('project_config').upsert({

                        id: 1,

                        workflow_text: config.value.workflow_text,

                        workflow_images: config.value.workflow_images,

                        schedules: config.value.schedules,

                        updated_at: new Date()

                    });

                    if (!error) alert('✅ 流程与排班配置同步成功！');

                    loading.value = false;

                };


                const uploadImage = async (e) => {

                    const file = e.target.files[0];

                    if (!file) return;

                    loading.value = true;

                    const fileName = `public/${Date.now()}_${file.name}`;

                    const { data, error } = await supabase.storage.from('images').upload(fileName, file);

                    if (error) {

                        alert('图片上传失败，请确认 Storage 桶策略已开启');

                    } else {

                        const { data: { publicUrl } } = supabase.storage.from('images').getPublicUrl(fileName);

                        config.value.workflow_images.push(publicUrl);

                    }

                    loading.value = false;

                };


                const openLog = (log = null) => {

                    if (log) {

                        editLog.value = JSON.parse(JSON.stringify(log));

                    } else {

                        editLog.value = { 

                            date: new Date().toISOString().split('T')[0], 

                            leader: '', 

                            is_finished: false, 

                            issues: [] 

                        };

                    }

                };


                const saveLog = async () => {

                    loading.value = true;

                    const { error } = await supabase.from('daily_logs').upsert(editLog.value);

                    if (!error) {

                        editLog.value = null;

                        await fetchData();

                    }

                    loading.value = false;

                };


                const syncIssueStatus = async (log) => {

                    await supabase.from('daily_logs').update({ issues: log.issues }).eq('id', log.id);

                };


                onMounted(fetchData);


                return {

                    tabs, currentTab, loading, config, logs, editLog,

                    saveConfig, uploadImage, openLog, saveLog, syncIssueStatus

                };

            }

        }).mount('#app');

    </script>
